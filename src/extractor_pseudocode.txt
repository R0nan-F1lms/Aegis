FUNCTION is_section_name_standard(name):
    # Checks against known good names like .text, .data, .rsrc
    IF name in STANDARD_NAMES:
        RETURN True
    ELSE:
        RETURN False

FUNCTION detect_packing(entropy, section_names):
    # High entropy (> 7.0) or strange section names often indicates a packer (like UPX)
    IF entropy > THRESHOLD OR non_standard_names_found:
        RETURN 1 # Packed
    ELSE:

# --- CORE EXTRACTION LOGIC ---

FUNCTION extract_features_from_pe(file_path, class_label):
    TRY:
        Load file_path into pefile.PE() object
        
        # Initialize an empty dictionary for this file's features
        file_features = {}
        
        # 1. Structural Features
        file_features["class_label"] = class_label
        file_features["number_of_sections"] = count PE sections
        
        # 2. Section Mismatches & Names
        mismatch_count = 0
        weird_name_count = 0
        FOR each section in PE:
            IF section.VirtualSize differs greatly from section.SizeOfRawData:
                mismatch_count += 1
            IF NOT is_section_name_standard(section.Name):
                weird_name_count += 1
                
        file_features["size_mismatch_count"] = mismatch_count
        file_features["weird_section_names"] = weird_name_count
        
        # 3. Payload Entropy
        payload_bytes = read main executable section
        file_features["payload_entropy"] = calculate_entropy(payload_bytes)
        
        # 4. Packer Detection
        file_features["is_packed"] = detect_packing(file_features["payload_entropy"])
        
        # 5. API & DLL Imports
        FOR each entry in PE.DIRECTORY_ENTRY_IMPORT:
            Add DLL name to file_features
            FOR each API function in entry:
                Add API name to file_features
                
        RETURN file_features
        
    CATCH pefile.PEFormatError:
        # Some malware is intentionally corrupted to break analysis tools
        Log error "Invalid PE format"
        RETURN None


# --- MAIN PIPELINE ---

FUNCTION main():
    Initialize empty list: all_extracted_data = []
    
    # Process Malware
    FOR each file in "/data/malware/":
        Determine class_label from folder name or file name
        features = extract_features_from_pe(file, class_label)
        IF features is valid:
            Append features to all_extracted_data
            
    # Process Benign
    FOR each file in "/data/benign/":
        features = extract_features_from_pe(file, "Benign")
        IF features is valid:
            Append features to all_extracted_data
            
    # Output to CSV
    Open "/output/features.csv" for writing
    Extract all unique column headers (all APIs, DLLs, and structural names)
    Write headers to CSV
    FOR each row in all_extracted_data:
        Write row to CSV (inserting '0' if an API was missing, '1' if present)
        
    PRINT "Extraction complete. Dataset ready for ML."

CALL main()
